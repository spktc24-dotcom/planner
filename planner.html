1Відкрий Google Apps Script

Перейди: script.google.com
Натисни «Новий проект» (кнопка зліва зверху)

2Встав цей код

Виділи весь старий код у редакторі (Ctrl+A), видали і встав цей:

const SPREADSHEET_ID = '1Y5YIuCSRnYJep9iWx6tpsWN3pzwBc4CoOWEL8awElKA'; const SHEET_NAME = 'Tasks'; function getSheet() { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SHEET_NAME); if (!sheet) { sheet = ss.insertSheet(SHEET_NAME); const h = sheet.getRange(1,1,1,7); h.setNumberFormat('@'); h.setValues([['id','date','timeStart','timeEnd','text','priority','done']]); } return sheet; } function handle(e) { try { let payload = null; if (e.parameter && e.parameter.payload) payload = e.parameter.payload; else if (e.postData && e.postData.contents) payload = e.postData.contents; if (!payload) return out({ok:false, error:'no payload'}); const data = JSON.parse(payload); const sheet = getSheet(); const action = data.action; if (action === 'getAll') { const rows = sheet.getDataRange().getValues(); const tasks = rows.slice(1) .filter(r => r[0] && String(r[0]).trim() !== '') .map(r => ({ id: String(r[0]), date: String(r[1]), timeStart: String(r[2]), timeEnd: String(r[3]), text: String(r[4]), priority: String(r[5]), done: String(r[6]) === 'true' })); return out({ok:true, tasks}); } if (action === 'save') { const t = data.task; const rows = sheet.getDataRange().getValues(); let found = -1; for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(t.id)) { found = i+1; break; } } const row = [String(t.id),String(t.date),String(t.timeStart||''), String(t.timeEnd||''),String(t.text),String(t.priority),String(t.done)]; if (found > 0) { const r = sheet.getRange(found,1,1,7); r.setNumberFormat('@'); r.setValues([row]); } else { const r = sheet.getRange(sheet.getLastRow()+1,1,1,7); r.setNumberFormat('@'); r.setValues([row]); } return out({ok:true}); } if (action === 'delete') { const rows = sheet.getDataRange().getValues(); for (let i = 1; i < rows.length; i++) { if (String(rows[i][0]) === String(data.id)) { sheet.deleteRow(i+1); break; } } return out({ok:true}); } return out({ok:false, error:'unknown action'}); } catch(err) { return out({ok:false, error:err.toString()}); } } function out(obj) { return ContentService.createTextOutput(JSON.stringify(obj)).setMimeType(ContentService.MimeType.JSON); } function doGet(e) { return handle(e); } function doPost(e) { return handle(e); }
3Збережи і розгорни

3. Тип: «Веб-застосунок»
4. «Виконувати від імені»: Мене
5. «Хто має доступ»: Усі
6. Натисни «Розгорнути» → дозволь доступ у спливаючому вікні
7. Скопіюй URL веб-застосунку
